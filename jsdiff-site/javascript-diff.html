Install via NPM

                
npm install diff

            

            

                
Compare JavaScript Code

                
import { diffLines } from 'diff';

const oldCode = 'function add(a, b) { return a + b; }';
const newCode = 'const add = (a, b) => a + b;';

const differences = diffLines(oldCode, newCode);

differences.forEach(part => {
  if (part.added) console.log('+', part.value);
  else if (part.removed) console.log('-', part.value);
});
